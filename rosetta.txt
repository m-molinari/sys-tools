# BASIC #
###############################################################################################################################
-------------------------------------------------------------------------------------------------------------------------------
Install a package(s) by name	pacman -S		dnf install		apt install		emerge [-a] 
-------------------------------------------------------------------------------------------------------------------------------
Remove a package(s) by name	pacman -Rs		dnf remove		apt remove		emerge -[a]vc
-------------------------------------------------------------------------------------------------------------------------------
Search for package(s) 		pacman -Ss		dnf search		apt search		emerge -S 
-------------------------------------------------------------------------------------------------------------------------------
Upgrade Packages		pacman -Syu		dnf upgrade		apt upgrade		emerge -[a]uDN @world
-------------------------------------------------------------------------------------------------------------------------------
Clean up all local caches	pacman -Scc		dnf clean all		apt autoclean		eclean distfiles
-------------------------------------------------------------------------------------------------------------------------------
Remove dep no longer need	pacman -Qdtq \		dnf autoremove		apt autoremove		emerge [-a] --depclean
				| pacman -Rs -
-------------------------------------------------------------------------------------------------------------------------------
Only downloads			pacman -Sw		dnf download		apt install \		emerge [-a] --fetchonly
										--download-only	 
-------------------------------------------------------------------------------------------------------------------------------
